### 第一种
```javascript
var twoSum = function(nums, target) {
    for (let i = 0; i < nums.length; i++) {
        let temp = target - nums[i];
        let index = nums.indexOf(temp, i + 1);
        if (index !== -1) {
            return [i, index]
        }
    }
}
```
时间复杂度O(n),空间复杂度1

### 第二种
```javascript
var twoSum = function(nums, target) {
    let map = new Map();
    for (let i = 0; i < nums.length; i++) {
        let temp = target - nums[i]
        if (map.get(temp) !== undefined) {
            return [map.get(temp), i];
        }
        map.set(nums[i], i)
    }
}
```

时间复杂度O(n),空间复杂度O(n)
