## 定义
一定在事件触发n秒之后才执行，如果在一个事件触发的n秒内又触发了这个事件，那么以新触发的事件的时间为准，n秒后才执行，==总之就是要等触发完事件n秒后不再触发该事件，才最终执行==。
## Note
- 解决this的问题
    
```
setTimeout(function() {
        func.apply(context)
    }, time)
```

- 解决参数的问题

```
setTimeout(function() {
        func.apply(context, args)
    }, time)
```
- 立即执行
```
if (immediate) {
    var callnow = !timeout;
    timeout = setTimeout(function() {
        timeout = null
    },time)
    if (callnow) func.apply(context, args)
}
```
- 返回值
不立即执行的时候返回值始终是undefined，所以只在立即执行的时候返回值才有意义。
- cancel取消防抖

## 最终实现

```
<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <title>target属性的取值和作用</title>
    <script src="https://cdn.staticfile.org/jquery/1.10.2/jquery.min.js"></script>
    <style>
        #container{
            width: 100%; height: 200px; line-height: 200px; text-align: center; color: #fff; background-color: #444; font-size: 30px;
        }
    </style>
</head>
<body>
<div id="container"></div>
<button id="button">点击取消debounce</button>
<script>
  var count = 1;
  var container = document.getElementById("container");
  function getUserAction() {
    container.innerHTML = count++;
  }
  function debounce(func, wait, immediate) {

    var timeout, result;

    var debounced = function () {
        var context = this;
        var args = arguments;

        if (timeout) clearTimeout(timeout);
        if (immediate) {
            // 如果已经执行过，不再执行
            var callNow = !timeout;
            timeout = setTimeout(function(){
                timeout = null;
            }, wait)
            if (callNow) result = func.apply(context, args)
        }
        else {
            timeout = setTimeout(function(){
                func.apply(context, args)
            }, wait);
        }
        return result;
    }
    
    debounced.cancel = function() {
      clearTimeout(timeout);
      timeout = null;
    }

    return debounced;
}
  var setAction = debounce(getUserAction, 10000, true);

  container.onmousemove = setAction;
  document.getElementById("button").addEventListener("click", function() {
    setAction.cancel();
  });
</script>
</body>
</html>
```
